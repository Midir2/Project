
# 1-ый триггер
CREATE OR REPLACE FUNCTION function1()
  RETURNS TRIGGER
  LANGUAGE PLPGSQL
  AS
$$
BEGIN
NEW.DATE = current_date;
RETURN NEW;
END;
$$

CREATE TRIGGER trigger_date
  BEFORE INSERT
  ON CONSIGNMENT
  FOR EACH ROW
  EXECUTE PROCEDURE function1()
